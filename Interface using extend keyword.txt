import java.util.Scanner;

// Base interface
interface Animal {
    void eat();
}

// Extended interface
interface Dog extends Animal {
    void bark();
}

// Class implementing the extended interface
class Labrador implements Dog {
    private String name;

    // Constructor to set dog's name
    Labrador(String name) {
        this.name = name;
    }

    public void eat() {
        System.out.println(name + " is eating dog food.");
    }

    public void bark() {
        System.out.println(name + " is barking loudly.");
    }
}

public class DogInterfaceDemo {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // Get dog's name from user
        System.out.print("Enter dog's name: ");
        String dogName = scanner.nextLine();

        Labrador lab = new Labrador(dogName);

        // Menu for user actions
        while (true) {
            System.out.println("\nChoose an action:");
            System.out.println("1. Eat");
            System.out.println("2. Bark");
            System.out.println("3. Exit");
            System.out.print("Enter choice: ");
            int choice = scanner.nextInt();

            switch (choice) {
                case 1:
                    lab.eat();
                    break;
                case 2:
                    lab.bark();
                    break;
                case 3:
                    System.out.println("Goodbye!");
                    scanner.close();
                    return;
                default:
                    System.out.println("Invalid choice. Try again.");
            }
        }
    }
}
