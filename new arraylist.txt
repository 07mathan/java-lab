package exam;

import java.util.*;

public class ArrayListVectorUserInput {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // --- i) ArrayList Example ---
        System.out.println("=== ArrayList Example ===");
        ArrayList<String> fruits = new ArrayList<>();

        System.out.print("Enter number of fruits: ");
        int n = sc.nextInt();
        sc.nextLine(); // consume newline

        for (int i = 0; i < n; i++) {
            System.out.print("Enter fruit " + (i + 1) + ": ");
            fruits.add(sc.nextLine());
        }

        System.out.println("Fruits: " + fruits);

        System.out.print("Enter fruit to remove: ");
        String rem = sc.nextLine();
        fruits.remove(rem);
        System.out.println("After removal: " + fruits);

        // --- ii) Vector Example ---
        System.out.println("\n=== Vector Example ===");
        Vector<Integer> numbers = new Vector<>();

        System.out.print("Enter number of integers: ");
        int m = sc.nextInt();

        for (int i = 0; i < m; i++) {
            System.out.print("Enter number " + (i + 1) + ": ");
            numbers.add(sc.nextInt());
        }

        System.out.println("Numbers: " + numbers);

        System.out.print("Enter index to remove (0-based): ");
        int idx = sc.nextInt();
        if (idx >= 0 && idx < numbers.size()) {
            numbers.remove(idx);
        } else {
            System.out.println("Invalid index!");
        }

        System.out.println("After removal: " + numbers);

        System.out.println("All numbers using Enumeration:");
        Enumeration<Integer> e = numbers.elements();
        while (e.hasMoreElements()) {
            System.out.println("- " + e.nextElement());
        }

        sc.close();
    }
}
